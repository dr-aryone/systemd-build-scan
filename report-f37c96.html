183static unsigned burst_modulate(unsigned burst, uint64_t available) {
184 unsigned k;
185
186 /* Modulates the burst rate a bit with the amount of available
187 * disk space */
188
189 k = u64log2(available);
190
191 /* 1MB */
192 if (k <= 20)
193 return burst;
194
195 burst = (burst * (k-20)) / 4;
196
197 /*
198 * Example:
199 *
200 * <= 1MB = rate * 1
201 * 16MB = rate * 2
202 * 256MB = rate * 3
203 * 4GB = rate * 4
204 * 64GB = rate * 5
205 * 1TB = rate * 6
206 */
207
208 return burst;
209}
