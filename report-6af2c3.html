314int open_terminal(const char *name, int mode) {
315 int fd, r;
316 unsigned c = 0;
317
318 /*
319 * If a TTY is in the process of being closed opening it might
320 * cause EIO. This is horribly awful, but unlikely to be
321 * changed in the kernel. Hence we work around this problem by
322 * retrying a couple of times.
323 *
324 * https://bugs.launchpad.net/ubuntu/+source/linux/+bug/554172/comments/245
325 */
326
327 if (mode & O_CREAT0100)
328 return -EINVAL22;
329
330 for (;;) {
331 fd = open(name, mode, 0);
332 if (fd >= 0)
333 break;
334
335 if (errno(*__errno_location ()) != EIO5)
336 return -errno(*__errno_location ());
337
338 /* Max 1s in total */
339 if (c >= 20)
340 return -errno(*__errno_location ());
341
342 usleep(50 * USEC_PER_MSEC((usec_t) 1000ULL));
343 c++;
344 }
345
346 r = isatty(fd);
347 if (r < 0) {
348 safe_close(fd);
349 return -errno(*__errno_location ());
350 }
351
352 if (!r) {
353 safe_close(fd);
354 return -ENOTTY25;
355 }
356
357 return fd;
358}
