640static int path_dispatch_io(sd_event_source *source, int fd, uint32_t revents, void *userdata) {
641 PathSpec *s = userdata;
642 Path *p;
643 int changed;
644
645 assert(s)do { if ((__builtin_expect(!!(!(s)),0))) log_assert_failed("s"
, "src/core/path.c", 645, __PRETTY_FUNCTION__); } while (0);
646 assert(s->unit)do { if ((__builtin_expect(!!(!(s->unit)),0))) log_assert_failed
("s->unit", "src/core/path.c", 646, __PRETTY_FUNCTION__); }
while (0);
647 assert(fd >= 0)do { if ((__builtin_expect(!!(!(fd >= 0)),0))) log_assert_failed

648
649 p = PATH(s->unit);
650
651 if (p->state != PATH_WAITING &&
652 p->state != PATH_RUNNING)
653 return 0;
654
655 /* log_debug("inotify wakeup on %s.", u->id); */
656
657 LIST_FOREACH(spec, s, p->specs)for ((s) = (p->specs); (s); (s) = (s)->spec_next)

, __func__, "Got event on unknown fd.") : -abs(_e); });

664 }

666 changed = path_spec_fd_event(s, revents);
667 if (changed < 0)

669
670 /* If we are already running, then remember that one event was
671 * dispatched so that we restart the service only if something
672 * actually changed on disk */

675 if (changed)
676 path_enter_running(p);
677 else

680 return 0;
681
682fail:
