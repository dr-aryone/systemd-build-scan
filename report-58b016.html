645int image_read_only(Image *i, bool_Bool b) {
646 _cleanup_release_lock_file___attribute__((cleanup(release_lock_file))) LockFile global_lock = LOCK_FILE_INIT{ .fd = -1, .path = ((void*)0) }, local_lock = LOCK_FILE_INIT{ .fd = -1, .path = ((void*)0) };
647 int r;
648 assert(i)do { if ((__builtin_expect(!!(!(i)),0))) log_assert_failed("i"

(0);
649

651 return -EROFS30;
652
653 /* Make sure we don't interfere with a running nspawn */
654 r = image_path_lock(i->path, LOCK_EX2|LOCK_NB4, &global_lock, &local_lock);
655 if (r < 0)
656 return r;
657
658 switch (i->type) {
659
660 case IMAGE_SUBVOLUME:
661
662 /* Note that we set the flag only on the top-level
663 * subvolume of the image. */
664
665 r = btrfs_subvol_set_read_only(i->path, b);
666 if (r < 0)
667 return r;
668
669 break;
670
671 case IMAGE_DIRECTORY:

673 mode of the top-level directory, since it has an
674 effect on the container itself. However, we can
675 use the "immutable" flag, to at least make the
676 top-level directory read-only. It's not as good as
677 a read-only subvolume, but at least something, and
678 we can read the value back.*/
679
680 r = chattr_path(i->path, b ? FS_IMMUTABLE_FL0x00000010 : 0, FS_IMMUTABLE_FL0x00000010);
681 if (r < 0)
682 return r;
683
684 break;

691

693 return -errno(*__errno_location ());

696 * defrag it, given that no write patterns will
697 * fragment it again. */

710int image_path_lock(const char *path, int operation, LockFile *global, LockFile *local) {
711 _cleanup_free___attribute__((cleanup(freep))) char *p = NULL((void*)0);
712 LockFile t = LOCK_FILE_INIT{ .fd = -1, .path = ((void*)0) };

716 assert(path)do { if ((__builtin_expect(!!(!(path)),0))) log_assert_failed

717 assert(global)do { if ((__builtin_expect(!!(!(global)),0))) log_assert_failed

735 }

738 if (r < 0)
739 return r;
740
741 if (p) {
742 mkdir_p("/run/systemd/nspawn/locks", 0700);
743
744 r = make_lock_file(p, operation, global);
745 if (r < 0) {
746 release_lock_file(&t);

748 }
749 }

751 *local = t;

753}
