633static int spawn_wait(struct udev_event *event,
634 usec_t timeout_usec,
635 usec_t timeout_warn_usec,

641 .accept_failure = accept_failure,

644 int r, ret;
645
646 r = sd_event_new(&e);

650 if (timeout_usec > 0) {

664 }
