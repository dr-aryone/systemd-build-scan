122void release_lock_file(LockFile *f) {
123 int r;
124
125 if (!f)
126 return;
127
128 if (f->path) {
129
130 /* If we are the exclusive owner we can safely delete
131 * the lock file itself. If we are not the exclusive
132 * owner, we can try becoming it. */
133
134 if (f->fd >= 0 &&
135 (f->operation & ~LOCK_NB4) == LOCK_SH1) {
136 static const struct flock fl = {
137 .l_type = F_WRLCK1,
138 .l_whence = SEEK_SET0,
139 };
140
141 r = fcntl(f->fd, F_OFD_SETLK37, &fl);
142 if (r < 0 && errno(*__errno_location ()) == EINVAL22)
143 r = flock(f->fd, LOCK_EX2|LOCK_NB4);
144
145 if (r >= 0)
146 f->operation = LOCK_EX2|LOCK_NB4;
147 }
148
149 if ((f->operation & ~LOCK_NB4) == LOCK_EX2)
150 unlink_noerrno(f->path);
151
152 f->path = mfree(f->path);
153 }
154
155 f->fd = safe_close(f->fd);
156 f->operation = 0;
