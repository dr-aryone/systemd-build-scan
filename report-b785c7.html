876static struct ps_struct *get_next_ps(struct ps_struct *ps, struct ps_struct *ps_first) {
877 /*
878 * walk the list of processes and return the next one to be
879 * painted
880 */
881 if (ps == ps_first)
882 return ps->next_ps;
883
884 /* go deep */
885 if (ps->children)
886 return ps->children;
887
888 /* find siblings */
889 if (ps->next)
890 return ps->next;
891
892 /* go back for parent siblings */
893 while (1) {
894 if (ps->parent && ps->parent->next)
895 return ps->parent->next;
896
897 ps = ps->parent;
898 if (!ps)
899 return ps;
900 }
901
902 return NULL((void*)0);
903}
